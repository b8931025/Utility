select p.IEDRTYP,r.iedrtyp,o.iedrtyp,a.iedrtyp,m.iedrtyp from pubm_pe_main  p
LEFT JOIN PUBM_PE_AREA R ON R.OID_PUBM_PE_MAIN = P.OID
LEFT JOIN PUBM_PE_OBJ o ON o.OID_PUBM_PE_MAIN = P.OID
LEFT JOIN PUBM_PE_ADDR a ON a.OID_PUBM_PE_MAIN = P.OID
LEFT JOIN PUBM_PE_PREM m ON m.OID_PUBM_PE_MAIN = P.OID
where 1=1
and p.iendors='0100FEN0010215';

select IREASON1,IREASON2,IREASON3,IREASON4,IREASON5
from FIRM_PE_MAIN f
where f.iendors = '0100FEN0010215';
