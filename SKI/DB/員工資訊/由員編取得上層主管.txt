SELECT ORGTBL1.IOFFICER 科主管代號,
			 ORGTBL1.NNAME 科主管姓名,
			 ORGTBL1.IEMAIL 科主管MAIL, 
       ORGTBL2.IOFFICER 部門主管代號,
       ORGTBL2.NNAME 部門主管姓名,
       ORGTBL2.IEMAIL 部門主管MAIL 
 FROM   PSNV_OFFICER_ALL OFFTBL 
 JOIN   PSNV_ORG3_NEW    ORGTBL1 ON OFFTBL.ISUBJNO = ORGTBL1.IUNITNO AND ORGTBL1.KORG IN ('3','4') 
 JOIN   PSNV_ORG3_NEW    ORGTBL2 ON OFFTBL.IDEPTNO = ORGTBL2.IUNITNO AND ORGTBL2.KORG = '2' 
 WHERE  OFFTBL.IOFFICER = '100245';
