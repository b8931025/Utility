OFF_NMAID(OFF_MAID(F.IOFFICER)||'00') AS NMANAGER,

FROM CMNP_OFF_SOURCE F, PSNV_PSN_ORG B              
WHERE F.IMANAGER = B.IOFFICER                       
AND (F.INVALID IS NULL OR  F.INVALID <> 'Y' )
